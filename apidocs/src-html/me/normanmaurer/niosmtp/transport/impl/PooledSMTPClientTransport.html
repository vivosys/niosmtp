017    package me.normanmaurer.niosmtp.transport.impl;

019    import java.net.InetSocketAddress;
020    import java.util.Iterator;
021    import java.util.concurrent.ConcurrentHashMap;
022    import java.util.concurrent.ConcurrentLinkedQueue;
023    import java.util.concurrent.Executors;
024    import java.util.concurrent.ScheduledExecutorService;
025    import java.util.concurrent.TimeUnit;
026    
027    import me.normanmaurer.niosmtp.SMTPResponse;
028    import me.normanmaurer.niosmtp.SMTPResponseCallback;
029    import me.normanmaurer.niosmtp.core.SMTPRequestImpl;
030    import me.normanmaurer.niosmtp.transport.SMTPClientConfig;
031    import me.normanmaurer.niosmtp.transport.SMTPClientSession;
032    import me.normanmaurer.niosmtp.transport.SMTPClientSession.CloseListener;
033    import me.normanmaurer.niosmtp.transport.SMTPClientTransport;
034    import me.normanmaurer.niosmtp.transport.SMTPDeliveryMode;

037    /**
038     * An {@link SMTPClientTransport} which tries to "pool" the {@link SMTPClientSession} for re-use. Use it with caution if
039     * you expect to connect to many different remote SMTP servers. This is best be used for the usage within the scope of a 
040     * mail relay.
041     * 
042     * @author Norman Maurer
043     *
044     */
045    public class PooledSMTPClientTransport extends LimitingSMTPClientTransport{
046    
047        private final static String WELCOME_RESPONSE_KEY = "WELCOME_RESPONSE";    
048        private final int keepAliveTimeInSec;
049        private final ConcurrentHashMap<String, ConcurrentLinkedQueue<PooledSMTPClientSession>> pooledSessions = new ConcurrentHashMap<String, ConcurrentLinkedQueue<PooledSMTPClientSession>>();
050        private final ScheduledExecutorService noopSender;
051        private final int keepAliveIntervalInSec;
052    
053        
054        private final SMTPResponseCallback noopSchedulingResponseCallback = new SMTPResponseCallback() {
055    
056            @Override
057            public void onResponse(final SMTPClientSession session, SMTPResponse response) {
058                PooledSMTPClientSession pooledSession = (PooledSMTPClientSession) session;
059                
060                // Check if the pooled session was not closed and is not in use atm
061                if (!pooledSession.isClosed() && !pooledSession.isInUse()) {
062                    
063                    // Check if the session is session can be used or if it was to long in the pool and so must get closed and removed
064                    if (pooledSession.getLastSent() < System.currentTimeMillis() - keepAliveTimeInSec) {
065                        noopSender.schedule(new Runnable() {
066                    
067                            @Override
068                            public void run() {
069                                session.send(SMTPRequestImpl.noop(), noopSchedulingResponseCallback);
070                            }
071                        }, keepAliveIntervalInSec, TimeUnit.SECONDS);
072                    } else {
073                        
074                        // make sure no other thread is closing it already
075                        if (pooledSession.acquire()) {
076                            pooledSession.send(SMTPRequestImpl.quit(), SMTPResponseCallback.EMPTY);
077                            pooledSession.close();
078                            pooledSession.getWrapped().close();
079                        }
080                    }
081                }
082            }
083    
084            @Override
085            public void onException(SMTPClientSession session, Throwable t) {
086                // close the session when receiving an Exception. 
087                PooledSMTPClientSession pooledSession = (PooledSMTPClientSession) session;
088                pooledSession.close();
089                pooledSession.getWrapped().close();
090            }
091            
092        };
093    
094        
095        public PooledSMTPClientTransport(SMTPClientTransport transport, int maxPoolSize, int keepAliveIntervalInSec, int keepAliveTimeInSec) {
096            this(transport, maxPoolSize, keepAliveIntervalInSec, keepAliveTimeInSec,  Executors.newScheduledThreadPool(10));
097        }
098        
099        public PooledSMTPClientTransport(SMTPClientTransport transport, int maxPoolSize, int keepAliveIntervalInSec, int keepAliveTimeInSec, ScheduledExecutorService noopSender) {
100            super(transport, maxPoolSize, -1);
101            if (transport.getDeliveryMode() == SMTPDeliveryMode.STARTTLS_DEPEND || transport.getDeliveryMode() == SMTPDeliveryMode.STARTTLS_TRY) {
102                throw new IllegalArgumentException("Pooled of starttls transport is not supported");
103            }
104            if (keepAliveIntervalInSec < keepAliveTimeInSec) {
105                throw new IllegalArgumentException("keepAliveIntervalInSec MUST be < keepAliveTimeInSec");
106            }
107            this.keepAliveTimeInSec = keepAliveTimeInSec;
108            this.keepAliveIntervalInSec = keepAliveIntervalInSec;
109            this.noopSender = noopSender;
110        }
111        
112        
113    
114        /**
115         * Return the maximal pool size
116         * 
117         * @return maxPoolSize
118         */
119        public int getMaxPoolSize() {
120            return getConnectionLimit();
121        }
122        
123        @Override
124        public void connect(InetSocketAddress remote, SMTPClientConfig config, final SMTPResponseCallback callback) {
125            String key = getKey(remote);
126            
127            // Get all pooled Sessions for the remote address
128            ConcurrentLinkedQueue<PooledSMTPClientSession> sessions = pooledSessions.get(key);
129            if (sessions == null) {
130                
131                // there were no pooled sessions yet. So create a "pool" for it and store it for later usage
132                sessions = new ConcurrentLinkedQueue<PooledSMTPClientSession>();
133                ConcurrentLinkedQueue<PooledSMTPClientSession> stored = pooledSessions.putIfAbsent(key, sessions);
134                if (stored != null) {
135                    sessions = stored;
136                }
137            }
138    
139            final ConcurrentLinkedQueue<PooledSMTPClientSession> pSessions = sessions;
140    
141            Iterator<PooledSMTPClientSession> pooledIt = pSessions.iterator();
142            PooledSMTPClientSession session = null;
143            
144            // iterate through the pool and check if there is a session left which we can use
145            while (pooledIt.hasNext()) {
146                session = pooledIt.next();
147                if (!session.isInUse() && session.acquire()) {
148                    
149                    // just write the cached welcome response back to the client.
150                    try {
151                        callback.onResponse(session, (SMTPResponse) session.getAttributes().get(WELCOME_RESPONSE_KEY));
152                    } catch (Exception e) {
153                        callback.onException(session, e);
154                    }
155                    break;
156                } else {
157                    if (!pooledIt.hasNext()) {
158                        // set the session to null if there are no more pooled sessions in place. 
159                        // This will make sure that we try to connect to the remote server and get a new one
160                        session = null;
161                    }
162                }
163            }
164            if (session == null) {
165                
166                // Connect to the remote server as we did not find a usable pooled session
167                super.connect(remote, config, new SMTPResponseCallback() {
168    
169                    @Override
170                    public void onResponse(SMTPClientSession session, SMTPResponse response) throws Exception {
171                        PooledSMTPClientSession pooledSession = new PooledSMTPClientSession(session);
172                        pooledSession.addCloseListener(new CloseListener() {
173    
174                            @Override
175                            public void onClose(final SMTPClientSession session) {
176                                if (!session.isClosed()) {
177                                    
178                                    // check if the session was closed for real. If not just send a RSET to reset it to a reusable state
179                                    session.send(SMTPRequestImpl.rset(), noopSchedulingResponseCallback);
180                                } else {
181                                    
182                                    // The session was really closed so remove it from the pooled sessions
183                                    pSessions.remove(session);
184    
185                                }
186                            }
187                        });
188                        
189                        // store the welcome response for later usage
190                        session.getAttributes().put(WELCOME_RESPONSE_KEY, response);
191                        callback.onResponse(pooledSession, response);
192                        pSessions.add(pooledSession);
193    
194                    }
195    
196                    @Override
197                    public void onException(SMTPClientSession session, Throwable t) {
198                        callback.onException(session, t);
199                    }
200                });
201            }
202    
203        }
204    
205        @Override
206        public void destroy() {
207            noopSender.shutdown();
208            
209            Iterator<ConcurrentLinkedQueue<PooledSMTPClientSession>> sessionsQueue = pooledSessions.values().iterator();
210            while(sessionsQueue.hasNext()) {
211                ConcurrentLinkedQueue<PooledSMTPClientSession> sessions = sessionsQueue.next();
212                PooledSMTPClientSession session = null;
213                while((session = sessions.poll()) != null) {
214                    session.getWrapped().close();
215                    session.close();
216                }
217            }
218          
219            super.destroy();
220            
221        }
222        
223        
224        private static String getKey(InetSocketAddress address) {
225            return address.getAddress().getHostAddress() + ":" + address.getPort();
226        }
