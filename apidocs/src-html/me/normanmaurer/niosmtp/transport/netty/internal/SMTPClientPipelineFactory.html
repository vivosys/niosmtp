017    package me.normanmaurer.niosmtp.transport.netty.internal;
018    
019    import me.normanmaurer.niosmtp.SMTPResponseCallback;
020    import me.normanmaurer.niosmtp.transport.SMTPClientConfig;
021    import me.normanmaurer.niosmtp.transport.SMTPDeliveryMode;
022    
023    import org.jboss.netty.channel.ChannelPipeline;
024    import org.jboss.netty.channel.ChannelPipelineFactory;
025    import org.jboss.netty.channel.Channels;
026    import org.jboss.netty.handler.codec.frame.DelimiterBasedFrameDecoder;
027    import org.jboss.netty.handler.codec.frame.Delimiters;
028    import org.jboss.netty.handler.stream.ChunkedWriteHandler;
029    import org.jboss.netty.handler.timeout.IdleStateHandler;
030    import org.jboss.netty.util.Timer;
031    import org.slf4j.Logger;
032    import org.slf4j.LoggerFactory;
033    
034    /**
035     * {@link ChannelPipelineFactory} which is used for the SMTP Client
036     * 
037     * @author Norman Maurer
038     * 
039     *
040     */
041    public class SMTPClientPipelineFactory implements ChannelPipelineFactory, NettyConstants{
042        protected final static Logger LOGGER = LoggerFactory.getLogger(SMTPClientPipelineFactory.class);
043        private final static DelimiterBasedFrameDecoder FRAMER = new DelimiterBasedFrameDecoder(8192, true, Delimiters.lineDelimiter());
044        private final static SMTPResponseDecoder SMTP_RESPONSE_DECODER = new SMTPResponseDecoder();
045        private final static SMTPRequestEncoder SMTP_REQUEST_ENCODER = new SMTPRequestEncoder();
046        private final static SMTPClientIdleHandler SMTP_CLIENT_IDLE_HANDLER = new SMTPClientIdleHandler();
047        private Timer timer;
048        protected SMTPResponseCallback callback;
049        protected SMTPClientConfig config;
050        
051        public SMTPClientPipelineFactory(SMTPResponseCallback callback, SMTPClientConfig config, Timer timer) {
052            this.timer = timer;
053            this.config = config;
054            this.callback = callback;
055        }
056        
057        
058        @Override
059        public ChannelPipeline getPipeline() throws Exception {
060            ChannelPipeline pipeline = Channels.pipeline();
061            pipeline.addLast(SMTP_IDLE_HANDLER_KEY, SMTP_CLIENT_IDLE_HANDLER);
062            pipeline.addLast(FRAMER_KEY, FRAMER);
063            pipeline.addLast(SMTP_RESPONSE_DECODER_KEY, SMTP_RESPONSE_DECODER);
064            pipeline.addLast(SMTP_REQUEST_ENCODER_KEY, SMTP_REQUEST_ENCODER);
065            pipeline.addLast(CHUNK_WRITE_HANDLER_KEY, new ChunkedWriteHandler());
066            
067            // Add the idle timeout handler
068            pipeline.addLast(IDLE_HANDLER_KEY, new IdleStateHandler(timer, 0, 0, config.getResponseTimeout()));
069            pipeline.addLast(CONNECT_HANDLER_KEY, createConnectHandler());
070            return pipeline;
071        }
072        
073        protected ConnectHandler createConnectHandler() {
074            return new ConnectHandler(callback, LOGGER, config, SMTPDeliveryMode.PLAIN, null);
075        }
076        
077    
078    
079    }
