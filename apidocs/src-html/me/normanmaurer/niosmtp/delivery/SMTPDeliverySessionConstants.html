026    /**
027     * Constants for temporary data which will get stored in the {@link SMTPClientSession#getAttributes()} while sending the message
028     * 
029     * @author Norman Maurer
030     *
031     */
032    public interface SMTPDeliverySessionConstants {
033    
034        /**
035         * Key under which the {@link SMTPDeliveryFutureImpl} is stored
036         */
037        public static final String FUTURE_KEY = "future";
038       
039        /**
040         * Key under which the current recipient (RCPT TO) is stored as a {@link String}
041         */
042        public static final String CURRENT_RCPT_KEY = "cur_rcpt";
043        
044        
045        /**
046         * Key under which the recipients (RCPT TO) is stored as a {@link Iterator}
047         */
048        public static final String RECIPIENTS_KEY = "recipients";
049        
050        /**
051         * Key under which the already processed {@link DeliveryRecipientStatus} are stored in a {@link List}
052         */
053        public static final String DELIVERY_STATUS_KEY = "deliverstatus";
054        
055        /**
056         * Key under which we store if <code>PIPELINING</code> is active. 
057         */
058        public final static String PIPELINING_ACTIVE_KEY = "pipelining_active";
059        
060        
061        /**
062         * Key under which the already processed {@link DeliveryResult}'s are stored as a {@link List}
063         */
064        public static final String DELIVERY_RESULT_LIST_KEY = "delivery_result_List";
065    
066        
067        
068        /**
069         * Key under which the {@link SMTPDeliveryEnvelope}'s are stored as a {@link Iterator}
070         */
071        public final static String SMTP_TRANSACTIONS_KEY = "smtp_transactions";
072        
073        /**
074         * Key under which the current {@link SMTPDeliveryEnvelope} is stored
075         */
076        public final static String CURRENT_SMTP_TRANSACTION_KEY = "cur_smtp_transaction";
077    
078    
079        /**
080         * Key under which the {@link SMTPResponseCallbackFactory} will be stored
081         */
082        public final static String SMTP_RESPONSE_CALLBACK_FACTORY ="smtp_response_callback_factory";
083    }
