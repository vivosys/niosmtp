017    package me.normanmaurer.niosmtp.delivery.callback;
018    
019    import java.util.Iterator;
020    import java.util.Set;
021    
022    import me.normanmaurer.niosmtp.SMTPException;
023    import me.normanmaurer.niosmtp.SMTPRequest;
024    import me.normanmaurer.niosmtp.SMTPResponse;
025    import me.normanmaurer.niosmtp.SMTPResponseCallback;
026    import me.normanmaurer.niosmtp.SMTPUnsupportedExtensionException;
027    import me.normanmaurer.niosmtp.core.SMTPRequestImpl;
028    import me.normanmaurer.niosmtp.delivery.Authentication;
029    import me.normanmaurer.niosmtp.delivery.SMTPDeliveryAgentConfig;
030    import me.normanmaurer.niosmtp.delivery.SMTPDeliveryAgentConfig.PipeliningMode;
031    import me.normanmaurer.niosmtp.delivery.SMTPDeliveryEnvelope;
032    import me.normanmaurer.niosmtp.transport.SMTPClientConstants;
033    import me.normanmaurer.niosmtp.transport.SMTPDeliveryMode;
034    import me.normanmaurer.niosmtp.transport.SMTPClientSession;
035    
036    /**
037     * {@link AbstractResponseCallback} implementation which will handle the <code>EHLO</code> {@link SMTPResponse}
038     * 
039     * It will write the next {@link SMTPRequest} to the {@link SMTPClientSession} with the right {@link SMTPResponseCallback} added.
040     * 
041     * This implementation also handles the <code>PIPELINING</code> and also the <code>STARTTLS</code> extension
042     * 
043     * @author Norman Maurer
044     *
045     */
046    public class EhloResponseCallback extends AbstractResponseCallback implements SMTPClientConstants{
047    
048        
049        /**
050         * Get instance of this {@link SMTPResponseCallback} implemenation
051         */
052        public static final SMTPResponseCallback INSTANCE = new EhloResponseCallback();
053        
054        private static final SMTPException PIPELINING_NOT_SUPPORTED_EXECTION = new SMTPUnsupportedExtensionException("Extension PIPELINING is not supported");
055        
056        private static final SMTPException STARTTLS_NOT_SUPPORTED_EXECTION = new SMTPUnsupportedExtensionException("Extension STARTTLS is not supported");
057    
058        private EhloResponseCallback() {
059            
060        }
061        
062        @Override
063        public void onResponse(SMTPClientSession session, SMTPResponse response) throws SMTPException {
064            boolean supportsPipelining = false;
065            boolean supportsStartTLS = false;
066            initSupportedExtensions(session, response);
067            
068            // Check if the SMTPServer supports PIPELINING 
069            Set<String> extensions = session.getSupportedExtensions();
070    
071            if (extensions.contains(PIPELINING_EXTENSION)) {
072                supportsPipelining = true;
073            }
074            if (extensions.contains(STARTTLS_EXTENSION)) {
075                supportsStartTLS = true;
076            }
077    
078            int code = response.getCode();
079    
080            String mail = ((SMTPDeliveryEnvelope)session.getAttributes().get(CURRENT_SMTP_TRANSACTION_KEY)).getSender();
081            if (code < 400) {
082    
083                // Check if we depend on pipelining 
084                if (!supportsPipelining && ((SMTPDeliveryAgentConfig)session.getConfig()).getPipeliningMode() == PipeliningMode.DEPEND) {
085                    throw  PIPELINING_NOT_SUPPORTED_EXECTION;
086                }
087    
088                if (!supportsStartTLS && session.getDeliveryMode() == SMTPDeliveryMode.STARTTLS_DEPEND) {
089                    throw STARTTLS_NOT_SUPPORTED_EXECTION;
090                }
091                
092                
093                
094                if (supportsStartTLS && (session.getDeliveryMode() == SMTPDeliveryMode.STARTTLS_DEPEND || session.getDeliveryMode() == SMTPDeliveryMode.STARTTLS_TRY)) {
095                    next(session, SMTPRequestImpl.startTls());
096                } else {
097                    Authentication auth = ((SMTPDeliveryAgentConfig)session.getConfig()).getAuthentication();
098                    if (auth == null) {
099                        // We use a SMTPPipelinedRequest if the SMTPServer supports
100                        // PIPELINING. This will allow the NETTY to get
101                        // the MAX throughput as the encoder will write it out in one
102                        // buffer if possible. This result in less system calls
103                        if (supportsPipelining && ((SMTPDeliveryAgentConfig)session.getConfig()).getPipeliningMode() != PipeliningMode.NO) {
104                            pipelining(session);
105                        } else {
106                            next(session, SMTPRequestImpl.mail(mail));
107                        }
108                    } else {
109                        switch (auth.getMode()) {
110                        case Plain:
111                            next(session, SMTPRequestImpl.authPlain());
112    
113                            break;
114                        case Login:
115                            next(session, SMTPRequestImpl.authLogin());
116                        default:
117                            break;
118                        }
119                    }
120                }
121    
122            } else {
123                setDeliveryStatusForAll(session, response);
124    
125            }
126            
127        }
128        
129        
130        
131        /**
132         * Return all supported extensions which are included in the {@link SMTPResponse}
133         * 
134         * @param response
135         * @return extensions
136         */
137        private void initSupportedExtensions(SMTPClientSession session, SMTPResponse response) {
138            Iterator<String> lines = response.getLines().iterator();
139            while(lines.hasNext()) {
140                String line = lines.next();
141                if (line.equalsIgnoreCase(PIPELINING_EXTENSION)) {
142                    session.addSupportedExtensions(PIPELINING_EXTENSION);
143                } else if (line.equalsIgnoreCase(STARTTLS_EXTENSION)) {
144                    session.addSupportedExtensions(STARTTLS_EXTENSION);
145                }
146            }
147        }
148        
149    }
