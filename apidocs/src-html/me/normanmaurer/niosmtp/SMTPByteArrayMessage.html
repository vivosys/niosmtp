017    package me.normanmaurer.niosmtp;
018    
019    import java.io.ByteArrayInputStream;
020    import java.io.IOException;
021    import java.io.InputStream;
022    
023    
024    /**
025     * Abstract base implementation of {@link SMTPMessage} which allows to access the raw <code>byte</code> array of the {@link SMTPMessage}. This 
026     * should only be used if you are sure that the whole {@link SMTPMessage} fit into memory. 
027     * 
028     * The {@link #get7BitAsByteArray()} and {@link #get8BitAsByteArray()} methods can be used by the Transport implementation for better performing
029     * write operations.
030     * 
031     * 
032     * @author Norman Maurer
033     *
034     */
035    public abstract class SMTPByteArrayMessage implements SMTPMessage{
036    
037        private InputStream _7bitIn;
038        private InputStream _8bitIn;
039        
040        
041        @Override
042        public InputStream get7bit() throws IOException {
043            if (_7bitIn == null) {
044                _7bitIn = new ByteArrayInputStream(get7BitAsByteArray());
045            }
046            return _7bitIn;
047        }
048    
049        @Override
050        public InputStream get8Bit() throws IOException {
051            if (_8bitIn == null) {
052                _8bitIn = new ByteArrayInputStream(get8BitAsByteArray());
053            }
054            return _8bitIn;
055        }
056        
057        /**
058         * Return the <code>byte</code> array which should be used for the 7Bit message input.
059         * 
060         * @return 7bit
061         */
062        public abstract byte[] get7BitAsByteArray();
063    
064        /**
065         * Return the <code>byte</code> array which should be used for the 8Bit message input.
066         * 
067         * @return 8bit
068         */
069        public abstract byte[] get8BitAsByteArray();
070    
071    }
