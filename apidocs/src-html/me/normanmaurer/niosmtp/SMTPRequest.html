017    package me.normanmaurer.niosmtp;
018    
019    /**
020     * A SMTP Request which is send to the SMTP-Server
021     * 
022     * @author Norman Maurer
023     *
024     */
025    public interface SMTPRequest {
026        
027        public final static String STARTTLS_COMMAND = "STARTTLS";
028        public final static String EHLO_COMMAND = "EHLO";
029        public final static String MAIL_COMMAND = "MAIL FROM:";
030        public final static String RCPT_COMMAND = "RCPT TO:";
031        public final static String AUTH_COMMAND = "AUTH";
032        public final static String AUTH_PLAIN_ARGUMENT = "PLAIN";
033        public final static String AUTH_LOGIN_ARGUMENT = "LOGIN";
034        public final static String DATA_COMMAND = "DATA";
035        public final static String QUIT_COMMAND = "QUIT";
036    
037        /**
038         * Separator which is used to separate the command from the argument
039         */
040        public final static char SEPERATOR = ' ';
041        
042        /**
043         * Return the command 
044         * 
045         * @return command
046         */
047        String getCommand();
048        
049        /**
050         * Return the argument. This may be null
051         * 
052         * @return argument
053         */
054        String getArgument();
055    }
