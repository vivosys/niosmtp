62          if (hook instanceof SimpleHook) {
63              hook = new SimpleHookAdapter((SimpleHook)hook);
64          }
65          chain.addHook(hook);
66          chain.addHook(new TestAuthHook());
67          return new NettyServer(new SMTPProtocol(chain, new LMTPConfigurationImpl()));
68      }

89      private void checkRejectOneRecipientAfterData(AssertCheck check) throws Exception {
90          int port = TestUtils.getFreePort();
91          
92  
93          NettyServer smtpServer = create(new DeliverToRecipientHook() {
94              
95              @Override
96              public HookResult deliver(SMTPSession session, MailAddress address, MailEnvelope env) {
97                  if (address.toString().equals("to@example.com")) {
98                      return new HookResult(HookReturnCode.DENY);
99                  }
100                 return new HookResult(HookReturnCode.OK);
101             }
102         });
103         
104         smtpServer.setListenAddresses(Arrays.asList(new InetSocketAddress(port)));
105 
106         smtpServer.bind();
107 
108 
109        
110         
111         SMTPClientTransport transport = createSMTPClient();
112         SMTPDeliveryAgent c = createAgent(transport);
113 
114         try {
115             SMTPDeliveryAgentConfigImpl conf = createConfig();
116             SMTPDeliveryEnvelope transaction = new SMTPDeliveryEnvelopeImpl("from@example.com", Arrays.asList(new String[] {"to@example.com", "to2@example.com", "to3@example.com"}), new SMTPMessageImpl(new ByteArrayInputStream("msg".getBytes())));
117             
118             SMTPDeliveryFuture future = c.deliver(new InetSocketAddress(port), conf,transaction);
119             check.onSMTPClientFuture(future);
120             
121         } finally {
122             smtpServer.unbind();
123             transport.destroy();
124         }
