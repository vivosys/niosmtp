17  package me.normanmaurer.niosmtp.core;
18  
19  import java.util.Iterator;
20  
21  import me.normanmaurer.niosmtp.SMTPMultiResponseCallback;
22  import me.normanmaurer.niosmtp.SMTPResponse;
23  import me.normanmaurer.niosmtp.SMTPResponseCallback;
24  import me.normanmaurer.niosmtp.transport.SMTPClientSession;
25  
26  /**
27   * {@link SMTPMultiResponseCallback} implementation which can be used for PIPELINING 
28   * 
29   * @author Norman Maurer
30   *
31   */
32  public class SMTPPipelingingResponseCallback implements SMTPMultiResponseCallback{
33  
34      private final static String CURRENT_CALLBACK_IN_USE = "CURRENT_CALLBACK_IN_USE";
35      
36      private final Iterator<SMTPResponseCallback> callbacks;
37  
38      public SMTPPipelingingResponseCallback(Iterator<SMTPResponseCallback> callbacks) {
39          this.callbacks = callbacks;
40      }
41      
42      
43      @Override
44      public void onResponse(SMTPClientSession session, SMTPResponse response) throws Exception {
45              
46          if (callbacks.hasNext()) {
47              SMTPResponseCallback callback = null;
48              try {
49                  callback = callbacks.next();
50                  callback.onResponse(session, response);
51              } catch (Exception e) {
52                  // set a marker so we know that we don't need to retrieve the next callback to handle the exception
53                  session.getAttributes().put(CURRENT_CALLBACK_IN_USE, callback);
54                  throw e;
55              }
56          }
57          
58          
59      }
60  
61      @Override
62      public void onException(SMTPClientSession session, Throwable t) {
63          SMTPResponseCallback callback = (SMTPResponseCallback) session.getAttributes().remove(CURRENT_CALLBACK_IN_USE);
64          if (callback == null) {
65              if (!isDone(session)) {
66                  callback = callbacks.next();
67              }
68          }
69          if (callback != null) {
70              callback.onException(session, t);
71          }
72  
73          
74          
75      }
76  
77      @Override
78      public boolean isDone(SMTPClientSession session) {
79          return !callbacks.hasNext();
80      }
81  
82  }
